//Средняя задача ДЗ (динамический двумерный массив 10 - 50)
#include <iostream>
#include <stdlib.h>

using namespace std;

void create_mas(int* Mas, int k, int n) {
    for (int i = 0; i < k; i++) {
        for (int j = 0; j < n; j++) {
            Mas[i * n + j] = rand() % 40 + 10;
        }
    }
    for (int i = 0; i < k; i++) {
        for (int j = 0; j < n; j++) {
            cout << Mas[i * n + j] << " ";
        }
        cout << " " << endl;
    }

}
int main() {
    int k, n;
    cin >> k;
    cin >> n;
    int* Mas = new int[k * n];
    create_mas(Mas, k, n);
}
