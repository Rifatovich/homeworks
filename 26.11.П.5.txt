//Прима
#include <iostream>
using namespace std;

int main() {
	int n, v1, v2, a, b, min_cost = 0;
	int mat[10][10];
	int mas[100] = { 0 };
	int mas_ind = 0;
	int vis[10] = { 0 };
	cout << "How many vertices?";
	cin >> n;
	cout << "Enter the adjacency matrix" << endl;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			cin >> mat[i][j];
			if (mat[i][j] == 0) mat[i][j] = 1000;
		}
	}
	vis[1] = 1;
	int k = 1, min = 1000;
	while (k < n) {
		for (int i = 0; i < n; i++) {
			for (int j = 1; j < n; j++) {
				if (mat[i][j] < min) {
					if (vis[i] != 0) {
						min = mat[i][j];
						v1 = a = i;
						v2 = b = j;
					}
					if (vis[v1] == 0 || vis[v2] == 0) {
						mas[mas_ind] = b;
						mas_ind++;
						k++;
						min_cost = min_cost + min;
						vis[b] = 1;
					}
				}
			}
		}
		mat[a][b] = mat[b][a] = 1000;
	}
	cout << 1 << "->";
	for (int i = 0; i < n - 1; i++) {
		cout << mas[i];
		if (i < n - 2) cout << "->";
	}
	cout << "Min cost" << min_cost;
	return 0;
}
