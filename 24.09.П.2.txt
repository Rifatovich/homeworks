#include <iostream>
#include<cmath>
using namespace std;

 
unsigned int SetBit (unsigned int a1, unsigned int a2)  //вводим функцию 
 
{
    
int result = 0;	

a2 = pow(2,a2); //возводим 2 в степень позиции бита

result = (a1|a2); //с помощью побитового или установливаем в i-тый бит единицу
  
return result;

}
